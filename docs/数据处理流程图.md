## 3. 详细求值流水线 — 以 `(add 3 4)` 为例

```mermaid
sequenceDiagram
    participant Reader as "read.c"
    participant Stack as "g_stack 值栈"
    participant Eval as "eval.c"
    participant EnvStack as "g_env_stack 环境栈"
    participant Print as "print.c"

    Reader->>Reader: read 读到 '('
    Reader->>Reader: read_list 递归解析
    Reader->>Reader: read_sym -> symbol add
    Reader->>Reader: read_int -> number 3
    Reader->>Reader: read_int -> number 4
    Reader->>Reader: pop_list 构建链表 (add 3 4)
    Reader->>Stack: push (add 3 4)

    Eval->>Stack: pop = (add 3 4)
    Eval->>Eval: eval_sexp(sexp false)
    Eval->>Eval: TYPE_LIST 分支
    Eval->>Stack: args = tail(sexp) = (3 4) 入栈
    Eval->>Eval: fun = eval(head) = symbol add

    Eval->>EnvStack: eval_sym 环境查找
    EnvStack-->>Eval: 未找到
    Eval->>Eval: 回退到全局 binding

    Eval->>Eval: fun 是 FN (x y) (+ x y)
    Eval->>Eval: apply 分支
    Eval->>Eval: funtype = FN 求值参数 3 4
    Eval->>EnvStack: prepare_env / env_push(3) env_push(x) / env_push(4) env_push(y)
    Eval->>Eval: tail_eval(+ x y)
    Eval->>Eval: eval(+) -> BUILTIN F_ADD
    Eval->>Eval: eval(x)->3 eval(y)->4
    Eval->>Eval: 3+4=7
    Eval->>Eval: tail_eval(7) -> 自求值 -> 7
    Eval->>Stack: restore_stack(ss)
    Eval->>EnvStack: env_restore_stack(ee)

    Eval->>Print: print(7)
    Print->>Print: TYPE_NUM 直接输出
    Print-->>用户: 7
```

### 8.1 eval_sexp 主循环完整结构


### 8.3 用户函数/宏应用 (apply)

```mermaid
flowchart TD
    APPLY_IN["funtype = head(fun)<br/>FN=fn MACRO=macro<br/>list = pop() 实参列表"] --> ASSERT{"funtype == FN 或 MACRO ?"}
    ASSERT -->|"否"| ERR["error: 不是函数"]
    ASSERT -->|"是"| SAVE["push(body) 保存函数体<br/>push(args) 保存形参列表"]

    SAVE --> EVAL_ARGS{"funtype == MACRO ?"}

    EVAL_ARGS -->|"是(宏)"| SKIP_EVAL["参数不求值<br/>直接使用原始参数"]
    EVAL_ARGS -->|"否(函数)"| DO_EVAL["逐个 eval 求值参数<br/>g_stack[i] = eval(g_stack[i])"]

    SKIP_EVAL --> RESTORE_ENV["env_restore_stack(ee)<br/>恢复调用者环境"]
    DO_EVAL --> RESTORE_ENV

    RESTORE_ENV --> PREPARE["prepare_env(args, ss0)<br/>形参与实参绑定到 env_stack<br/>处理固定参数 / & rest / 裸符号"]
    PREPARE --> POP_BODY["args = pop() 弹出形参<br/>body = pop() 弹出函数体"]

    POP_BODY --> PUSH_BODY["push_reverse_list(body)<br/>函数体表达式逆序入栈"]

    PUSH_BODY --> LOOP{"g_sp > ss ?"}

    LOOP -->|"是"| LAST{"是否为最后一个表达式?"}
    LAST -->|"否 中间表达式"| EVAL_DISCARD["res = eval(pop)<br/>求值但丢弃结果"]
    EVAL_DISCARD --> LOOP

    LAST -->|"是 最后一个表达式"| TAIL_CTRL{"判断 tail_macro 如何设置"}

    TAIL_CTRL -->|"funtype == MACRO"| MACRO_TAIL["tail_macro += 2<br/>宏展开增加两层尾调用"]
    TAIL_CTRL -->|"funtype == FN && tail_macro>0"| FN_TAIL["tail_macro++<br/>函数增加一层尾调用"]
    TAIL_CTRL -->|"funtype == FN && tail_macro==0"| FN_NOT_TAIL["tail_macro 不变<br/>非尾调用上下文"]

    MACRO_TAIL --> TAIL_EVAL["tail_eval(pop)<br/>sexp = pop()<br/>restore_stack(ss)<br/>goto eval_top"]
    FN_TAIL --> TAIL_EVAL
    FN_NOT_TAIL --> TAIL_EVAL

    LOOP -->|"否 所有表达式处理完毕"| GOTO_END["goto end"]
```

### 8.5 copy_body 详细流程 (深拷贝+闭包捕获)

```mermaid
flowchart TD
    CB_IN["copy_body(body)"] --> CB_SWITCH{"type_of(body)"}

    CB_SWITCH -->|"TYPE_LIST"| CHECK_EMPTY{"body == EMPTY_LIST ?"}
    CHECK_EMPTY -->|"是"| RET_BODY["return body"]

    CHECK_EMPTY -->|"否"| CHECK_QUOTE{"head(body) == QUOTE ?"}
    CHECK_QUOTE -->|"是"| RET_BODY

    CHECK_QUOTE -->|"否"| CHECK_MACRO{"head(body) 是符号且<br/>其值是 MACRO 类型?"}
    CHECK_MACRO -->|"是 (内联宏展开)"| EXPAND["body = expand(body)<br/>展开宏<br/>return copy_body(body) 递归处理"]
    CHECK_MACRO -->|"否"| CHECK_ENV{"g_env_sp == 0 ?"}

    CHECK_ENV -->|"是 (无环境绑定)"| RET_BODY["return body<br/>优化: 避免不必要拷贝"]
    CHECK_ENV -->|"否"| PUSH_LIST["push_list(body)<br/>将列表元素逐个入栈"]

    PUSH_LIST --> RECURSE["for each element:<br/>g_stack[i] = copy_body(g_stack[i])<br/>递归深拷贝每个子元素"]
    RECURSE --> POP_LIST["return pop_list(ss)<br/>重建列表"]

    CB_SWITCH -->|"TYPE_SYM"| CB_SYM["遍历 env_stack<br/>从 g_env_sp-1 向下步进 2"]
    CB_SYM --> FOUND{"找到符号?"}
    FOUND -->|"是"| RET_VAL["return g_env_stack[i-1]<br/>返回该符号当前绑定的值<br/>自由变量→值"]
    FOUND -->|"否"| RET_SELF["return body<br/>全局符号/未绑定符号保持不变"]

    CB_SWITCH -->|"default (number等)"| RET_SELF
```

### 8.6 cond 特殊形式详解

```mermaid
flowchart TD
    COND_IN["args 为 cond 子句列表<br/>((条件1 表达式1) (条件2 表达式2) ...)"] --> PUSH["push_list(args)<br/>子句入栈"]

    PUSH --> LOOP{"i 从 ss 遍历到 g_sp ?"}

    GET_COND --> PUSH_EXPR["push(head(tail_(pair)))<br/>结果表达式入栈"]

    EVAL_COND --> BOOL{"to_bool(res) != NIL ?"}

    BOOL -->|"是 条件成立"| TAIL_COND{"tail_macro > 0 ?"}
    TAIL_COND -->|"是"| INC_TAIL["++tail_macro<br/>增加尾调用上下文"]
    TAIL_COND -->|"否"| SKIP_INC["不增加"]
    INC_TAIL --> TAIL_EVAL["tail_eval(pop)<br/>求值对应表达式<br/>goto eval_top"]
    SKIP_INC --> TAIL_EVAL

    BOOL -->|"否 条件不成立"| POP_EXPR["pop() 丢弃表达式"]
    POP_EXPR --> LOOP

    LOOP -->|"无更多子句"| END["goto end<br/>res 保持最后一次求值结果"]
```

### 8.9 prepare_env 参数绑定详细流程

```mermaid
flowchart TD
    PE_IN["prepare_env(args, ss)<br/>args: 形参列表<br/>ss: 实参起始位置"] --> IS_LIST{"is_list(args) ?"}

    IS_LIST -->|"是 列表形参"| ITER["遍历形参列表 h<br/>for h = args; h != EMPTY_LIST; h = tail(h)"]

    ITER --> CHECK_REST{"head(h) == REST (&) ?"}

    CHECK_REST -->|"否 普通参数"| CHECK_SS{"ss == g_sp ?"}
    CHECK_SS -->|"是"| ERR["error: 参数不足"]
    CHECK_SS -->|"否"| BIND["env_push(g_stack[ss]) 实参值<br/>env_push(head(h)) 形参名<br/>++ss"]

    BIND --> ITER

    CHECK_REST -->|"是 & rest 参数"| REST_BRANCH["args = head(tail(h))<br/>取出 rest 参数的符号名<br/>break 退出循环"]

    REST_BRANCH --> REST_BIND["... 进入下面的 rest 处理"]

    IS_LIST -->|"否 裸符号变参"| VAR_ARG_BIND["env_push(pop_list(ss))<br/>所有剩余实参构成列表<br/>env_push(args) 绑定到裸符号"]

    ITER -->|"遍历结束"| CHECK_REST2{"args 不是列表<br/>(rest 或裸符号)?"}
    CHECK_REST2 -->|"是"| REST_VAR_BIND["env_push(pop_list(ss))<br/>剩余实参打包为列表<br/>env_push(args) 绑定到 &rest 符号"]
    CHECK_REST2 -->|"否 普通固定参数"| DONE["完成"]

    REST_VAR_BIND --> DONE
    VAR_ARG_BIND --> DONE
    DONE --> RESTORE["restore_stack(sp)<br/>清理值栈"]
```

### 8.11 完整求值链路示例 — `((fn (x) (add x 1)) 41)`


## 关键设计要点

| 阶段 | 核心文件 | 关键机制 | 数据格式 |
|------|----------|----------|----------|
| **读取** | `lisp_read.c` | 递归下降解析器, 语法糖展开 | 字符流 -> tagged value (AST) |
| **求值** | `lisp_eval.c` | goto 主循环, 尾调用优化, 深拷贝闭包 | tagged value -> tagged value |
| **环境** | `lisp_eval.c` | 环境栈交替存储, prepare_env 绑定 | value/symbol 配对栈 |
| **内置** | `lisp_eval.c` | X-macro 枚举27个, switch 分发 | Builtin {type code} |
| **打印** | `lisp_print.c` | 类型分发, 递归列表打印 | tagged value -> 字符串 |
| **GC** | `lisp_core.c` | Cheney 停止复制, heap *1.5 扩容 | from-space -> to-space |
| **符号** | `lisp_symbol.c` | BST 二叉搜索树, 哈希键 | Symbol 节点 |
